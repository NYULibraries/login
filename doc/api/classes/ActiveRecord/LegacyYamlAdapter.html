def self.convert(klass, coder)
  return coder unless coder.is_a?(Psych::Coder)

  case coder["active_record_yaml_version"]
  when 0 then coder
  else
    if coder["attributes"].is_a?(AttributeSet)
      coder
    else
      Rails41.convert(klass, coder)
    end
  end
end
